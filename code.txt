import pandas as pd

# Inputs
L1 = ['A', 'B', 'C']
L2 = ['G', 'H', 'I', 'J']
entity_name = 'MyEntity'

# Prepare data rows
rows = []
for bl in L1:
    for at in L2:
        row = {
            'entity': entity_name,
            'bl': bl,
            'at': at,
            'Open': 0,
            '2024': 0,
            '2025': 0
        }
        rows.append(row)

# Create DataFrame
df = pd.DataFrame(rows, columns=['entity', 'bl', 'at', 'Open', '2024', '2025'])

print(df)
