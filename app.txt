tax_calculation = {}

for cgu, years in pbt_dict.items():
    tax_calculation[cgu] = {}
    for year, pbt in years.items():
        # Get the tax rate for the year, fallback to Opening if year not found
        tax_rate = tax_dict.get(year, tax_dict['Opening']).get('tax rate', 0)
        tax_calculation[cgu][year] = pbt * tax_rate
