import pandas as pd
import re

# Load Excel file
file_path = 'your_file.xlsx'
xls = pd.ExcelFile(file_path)

# Dictionary to hold DataFrames
sheet_dfs = {}

for sheet_name in xls.sheet_names:
    # Replace non-alphanumeric characters with underscores
    safe_name = re.sub(r'\W|^(?=\d)', '_', sheet_name)
    # Read the sheet into a DataFrame
    sheet_dfs[safe_name] = pd.read_excel(xls, sheet_name=sheet_name)

# Optional: assign each df as a variable dynamically (not recommended for most use cases)
globals().update(sheet_dfs)
