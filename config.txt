import time
import pandas as pd
import webbrowser
from threading import Timer

from dash import Dash, html, dcc, dash_table, Input, Output, State

# -----------------------
# Dummy data
# -----------------------
df = pd.DataFrame(
    {
        "category": ["A"] * 30 + ["B"] * 30 + ["C"] * 30,
        "sub_category": (
            ["A1"] * 10 + ["A2"] * 10 + ["A3"] * 10
            + ["B1"] * 10 + ["B2"] * 10 + ["B3"] * 10
            + ["C1"] * 10 + ["C2"] * 10 + ["C3"] * 10
        ),
        "value": list(range(1, 91)),
    }
)

CATEGORY_OPTIONS = [{"label": c, "value": c} for c in sorted(df["category"].unique())]


def subcat_options_for(category: str):
    if not category:
        return []
    subcats = sorted(df.loc[df["category"] == category, "sub_category"].unique())
    return [{"label": s, "value": s} for s in subcats]


def filtered_rows(category: str, subcat: str):
    if not category or not subcat:
        return []
    dff = df[(df["category"] == category) & (df["sub_category"] == subcat)].copy()
    return dff.to_dict("records")


# -----------------------
# App
# -----------------------
app = Dash(__name__)

# CSS via index_string (works across versions)
app.index_string = """

    </style>
  </head>
  <body>
    {%app_entry%}
    <footer>

      {%scripts%}
      {%renderer%}
    </footer>
  </body>
</html>
"""

    "display": "none",  # will be toggled by callback running=
    "position": "fixed",
    "inset": "0",
    "backgroundColor": "rgba(0,0,0,0.55)",
    "zIndex": "9999",

    "justifyContent": "center",
    "flexDirection": "column",
    "gap": "14px",
}

app.layout = html.Div(
    style={"fontFamily": "system-ui, -apple-system, Segoe UI, Roboto, Arial", "padding": "18px"},
    children=[
        # Fullscreen overlay
        html.Div(

            style=OVERLAY_STYLE,

                html.Div(
                    style={
                        "width": "64px",
                        "height": "64px",

                        "borderTop": "7px solid white",

                    }
                ),
                html.Div(
                    "please wait while loading the data",
                    style={"color": "white", "fontSize": "18px", "fontWeight": "600"},
                ),
            ],
        ),

        html.H2("Dash Dashboard (Dummy Data)", style={"margin": "0 0 12px 0"}),

        html.Div(

# -----------------------
# Callbacks
# -----------------------

# dd1 -> dd2 options + reset dd2 + disable/enable + clear table
@app.callback(
    Output("dd2", "options"),
    Output("dd2", "value"),
    Output("dd2", "disabled"),
    Output("data_table", "data"),
    Input("dd1", "value"),
)
def update_dd2(category):
    if not category:
        return [], None, True, []
    opts = subcat_options_for(category)
    return opts, None, False, []  # dd2 stays blank until user selects


# dd2 -> load table (with lag) + fullscreen overlay while running
@app.callback(
    Output("data_table", "data"),
    Input("dd2", "value"),
    State("dd1", "value"),
    prevent_initial_call=True,
    running=[
        (Output("overlay", "style"), {**OVERLAY_STYLE, "display": "flex"}, {**OVERLAY_STYLE, "display": "none"})
    ],
)
def load_data(subcat, category):
    if not category or not subcat:
        return []
    time.sleep(2.8)  # lag so overlay stays visible
    return filtered_rows(category, subcat)


# -----------------------
# Auto-open browser
# -----------------------
def open_browser():
    webbrowser.open_new("http://127.0.0.1:8050/")


if __name__ == "__main__":
    Timer(1, open_browser).start()
    app.run_server(debug=True)
