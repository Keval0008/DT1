import dash
from dash import html, dcc

from utils.constants import APP_TITLE, TAB_REFERENCE_LABEL, TAB_FINANCIAL_LABEL


app = dash.Dash(__name__, title=APP_TITLE)
server = app.server


app.layout = html.Div(
    children=[
        html.H2(APP_TITLE),

        dcc.Tabs(
            id="main_tabs",
            value="reference_tab",
            children=[
                dcc.Tab(
                    label=TAB_REFERENCE_LABEL,
                    value="reference_tab",
                ),
                dcc.Tab(
                    label=TAB_FINANCIAL_LABEL,
                    value="financial_tab",
                ),
            ],
        ),

        html.Div(id="tab_content"),
    ]
)


@app.callback(
    dash.Output("tab_content", "children"),
    dash.Input("main_tabs", "value"),
)
def render_tab(tab_value):
    if tab_value == "reference_tab":
        return html.Div("Reference Data Check UI will go here")

    if tab_value == "financial_tab":
        return html.Div("Financial Data Check UI will go here")

    return html.Div()


if __name__ == "__main__":
    app.run_server(debug=True)
